#include <stdio.h>

#define Tamanho 5

int Soma(int array[], int tamanho) {
    int soma = 0;
    
    for (int i = 0; i < tamanho; i++) {
        soma += array[i];
    }
    
    return soma;
}

int main() {
    int nota[Tamanho];
    
    printf("Digite as 5 notas:\n");
    
    for (int i = 0; i < Tamanho; i++) {
        printf("Nota %d: ", i + 1);
        scanf("%d", &nota[i]);
    }
    
    int soma = Soma(nota, Tamanho);
    
    printf("A soma das notas Ã©: %d\n", soma);
    
    return 0;
}
