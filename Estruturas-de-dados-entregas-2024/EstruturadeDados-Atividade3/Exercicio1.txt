#include <stdio.h>

#define Tamanho 5

int MaiorElemento(int array[], int tamanho) {
    int maior = array[0]; 
    
    for (int i = 1; i < tamanho; i++) {
        if (array[i] > maior) {
            maior = array[i];
        }
    }
    
    return maior;
}

int main() {
    int nota[Tamanho];
    
    printf("Digite as 5 notas:\n");
    
    for (int i = 0; i < Tamanho; i++) {
        printf("Nota %d: ", i + 1);
        scanf("%d", &nota[i]);
    }
    
    int MaiorNota = MaiorElemento(nota, Tamanho);
    
    printf("A maior nota Ã©: %d\n", MaiorNota);
    
    return 0;
}
